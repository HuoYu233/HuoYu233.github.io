前向传播

从输入初步传递到输出，即为前向传播

一般实现

def dense(a_in, W, b, g):
	units = W.shape[1]
	a_out = np.zeros(units)
	for j in range(units):
		w = W[:, j]
		z = np.dot(w, a_in) + b
		a_out[j] = g(z)
	return a_out
def sequential(x):
    a1 = dense(x, W1, b1)
    a2 = dense(a1, W2, b2)
    a3 = dense(a2, W3, b3)
    f_x = a3
    return f_x
