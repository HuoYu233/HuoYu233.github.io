public static void quick_sort(int q[], int l, int r){
    if (l >= r) return;
    int i = l - 1, j = r + 1, x = q[l + r >> 1];
    while (i < j){
        do i ++ ; while (q[i] < x);
        do j -- ; while (q[j] > x);
        if (i < j) {
            int t = q[i];
            q[i] = q[j];
            q[j] = t;           
        }
    }
    quick_sort(q, l, j);
    quick_sort(q, j + 1, r);
}

public static void merge_sort(int q[], int l, int r){
    if (l >= r) return;
    int mid = l + r >> 1;

    merge_sort(q, l, mid);
    merge_sort(q, mid + 1, r);

    while (i <= mid && j <= r)

    else tmp[k ++ ] = q[j ++ ];

    while (i <= mid) tmp[k ++ ] = q[i ++ ];

}
